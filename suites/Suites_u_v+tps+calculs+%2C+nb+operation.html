def temps(f,L):
    tps=[]# contiendra une liste de temps de calcul
    for j in L:
        start = time.time()
        f(j)
        dureeCalcul=time.time() - start # dureeCalcul=temps de calcul de f(j)
        tps.append(dureeCalcul)
    return tps

X=range(10,23) 
 
Yu=temps(u,X)# liste des temps de calcul de u(10), u(11), u(12), ..., u(22)
Yv=temps(v,X)# liste des temps de calcul de v(10), v(11), v(12), ..., v(22)
axis([min(X),max(X),-0.05*(max(Yu)-min(Yu)),max(Yu)])# paramtres d'ouverture pour fentre graphique
plot(X,Yu,'r--') #   rouge tiret
plot(X,Yv,'b*')#   bleu toil
show()

def NbOpU(n):
    l=[0]
    def u(n):
        if n==0 : 
            return 1
        else :
            y=u(n-1)+u(n-1)
            l[0]+=1
            return y
    u(n)
    return l[0]
    
    
def NbOpV(n):
    l=[0]
    def v(n):
        if n==0 : 
            return 1
        else :
            y=2*v(n-1)
            l[0]+=1
            return y
    v(n)
    return l[0]

import timeit
def tempsAvecCalculsRepetes(L):
    tps=[]# contiendra une liste de temps de calcul
    for j in L:
        timeit.timeit(number=100000000)
        t = timeit.Timer(stmt="v(%d)"%j, setup="from __main__ import v")  
        tps.append(t.timeit() )
    return tps
    
    
X=range(10,22) 
Y=tempsAvecCalculsRepetes(X) 
plot(X,Y,'b*')
show()
